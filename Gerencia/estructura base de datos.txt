CREATE TABLE empleados(
    fecha_ingreso date NOT null,
    fecha_retiro date,
    sueldo double,
    idCarnet int ,
    nombres varchar(50),
    apellidos varchar(50),
    cedula bigint NOT null,
    telefono varchar(50),
    direccion varchar(50),
    id_cargo bigint NOT null,
    PRIMARY key (cedula),
    FOREIGN KEY(id_cargo) REFERENCES cargos(id)
);
